184 
185  if(NULL == windows->texture){
186  fprintf(stderr, "error SDL_CreateTexture: %s", SDL_GetError());
187  return EXIT_FAILURE;
188  }
189 
190  get_Tittle (windows->Type , windows->title);
191  SDL_SetWindowTitle(windows->window, windows->title);
192  SDL_Surface *surface = NULL;
193  SDL_Texture *texture = NULL;
194  surface = IMG_Load(filename);

201  IMG_Quit();
202  return EXIT_FAILURE;
203  }

205  SDL_FreeSurface((SDL_Surface *) surface); /* On libère la surface, on n’en a plus besoin */
206  if (NULL == texture) {
207  fprintf(stderr, "Erreur SDL_CreateTextureFromSurface: %s", SDL_GetError());
208  SDL_FreeSurface(surface); /* On libère la surface, on n’en a plus besoin */

213  SDL_DestroyTexture(windows->texture);
214  SDL_Quit();
215  IMG_Quit();
216  return EXIT_FAILURE;
217  }
218  SDL_SetRenderTarget(windows->renderer,windows->texture);
219  SDL_RenderCopy(windows->renderer,texture,NULL,NULL);
